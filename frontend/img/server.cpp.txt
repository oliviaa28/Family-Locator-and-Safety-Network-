char* upper( char *cuvant){
    int i=0;
    char c;
    while( cuvant[i] ){
        c=cuvant[i];
        cuvant[i]=toupper((unsigned char)c);
        i++;
    }
    return cuvant;
}

void eliminare_newline(char *cuvant ){
    char *pozitie= strchr(cuvant , '\n');

        *pozitie= '\0';
}

void trim( char *cuvant){ //doar de la inceput sau final 
   char *f; // final
   int lg= strlen(cuvant) -1 ;
    
   while( isspace((unsigned char)*cuvant) ) // de la inceput
         cuvant++;

    if(*cuvant == 0 ) return; 
    
    f= cuvant+lg;
    while( cuvant < f && issspace((unsigned char)*f )) f--;
    f[1]= '\0';
}

void initializare(){
   if((sd = socket(AF_INET, SOCK_STREAM, 0)) < 0) //creeare socket
        HANDLE_ERROR("[SERVER] Eroare la socket");

    setsockopt(sd, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(optval));
    bzero (&server, sizeof(server));

    server.sin_family= AF_INET;
    server.sin_addr.s_addr = htonl(INADDR_ANY);
    server.sin_port= htons(PORT);

    //atasare socket 
    if( bind(sd, (struct sockaddr *) &server, sizeof(struct sockaddr)) ==-1)
       HANDLE_ERROR("[SERVER] Eroare la bind. \n");
    
    //punem serverul sa asculte daca vin clienti pt conecatre 
    if( listen(sd, MAX_CLIENTI) == -1)
        HANDLE_ERROR("[SERVER] Eroare la functia listen \n");
    
    FD_ZERO (&active_fds);
    FD_SET (sd, &active_fds);         //adaugam socketul pe care tocmai l.am creeat

    tv.tv_sec =1;  tv.tv_usec=0;     //asteptam o secunda 

    max_fds = sd;                   //numarul maixim de descriptori folositi   

    initializare_clienti(); 

}

void servire_clienti()
{
    while(1){
        //ajustam multimea descriptorilor care sunt utilizati (copiem in read_fds)
        bcopy( (char *) &active_fds, (char *) &read_fds, sizeof(read_fds)); 

        if( select (max_fds+1, &read_fds, NULL, NULL, &tv) < 0)
           HANDLE_ERROR("[SERVER] Eroare la functia select. \n");

        //verificam daaca socketul e pregatit pt a accepta clienti 
        if( FD_ISSET (sd, &read_fds)){
            lg = sizeof(from);
            bzero(&from, sizeof(from));
           //acceptam un client
            client = accept( sd, (struct sockaddr *) &from, &lg);
            if(client < 0 )
               HANDLE_ERROR("[SERVER] Eroare la accept \n");
            if (max_fds < client)
                max_fds=client;   

            FD_SET(client, &active_fds); //adaugam noul socket in multimea descriptorilor activi
            adauga_client(client);
            
             // printf("[server] S-a conectat clientul cu descriptorul %d, de la adresa %s.\n",client, conv_addr (from));
	        //  fflush (stdout);
             }

         for( fd =0; fd<=max_fds; fd++){ //indiferent daca s-a conectat un client sau nu 
            if(fd != sd && FD_ISSET(fd, &read_fds)){
                //citire   mesaj de la clirnt 
                citire_mesaj(fd); //in interior l acitire avem si procesare 
            }
     }
  }
}

int cauta_client(int fd){
int ok=0, pozitie;
    for( int i=0; i<MAX_CLIENTI; i++ ){
       if( clt[i].fd == fd){ //am gasit clientul cu care lucram momentan 
        ok=1;
        pozitie=i;
        break;
       }
    }

   if( ok == 0){
    printf("[SERVER | cauta client] Eroare...client necunoscut. \n");
    return -1 ;} 

   return pozitie;
}
